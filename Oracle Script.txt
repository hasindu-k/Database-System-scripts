DEPT and EMP Relationship 


CREATE TYPE dept_t AS OBJECT(
dno CHAR(5),
dname VARCHAR2(20)
)
/

CREATE TYPE emp_t AS OBJECT(
eno CHAR(5),
ename VARCHAR2(20),
addr VARCHAR2(20),
workdept REF dept_t
)
/

CREATE TABLE dept OF dept_t
(
PRIMARY KEY(dno)
)
/

CREATE TABLE emp OF emp_t
(
PRIMARY KEY(eno)
)
/
